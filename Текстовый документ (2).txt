#include "colors.inc"

camera {
    location <20, 15, 25>
    look_at <5, 7, 0>
    angle 50
}

background { color rgb<0.2, 0.4, 0.8> }
light_source { <50, 100, -50> color rgb 1 }
light_source { <-20, 30, 40> color rgb 0.5 } // Добавим подсветку

plane {
    y, 0
    pigment { checker color Black, color Green scale 10 }
}

// ----------------------------------------------------------
// МАКРОС: Бутылка с заданным цветом сиропа
// Параметр: SyrupColor - цвет жидкости внутри
// ----------------------------------------------------------
#macro Bottle(SyrupColor)
union {
    // ----- ВНЕШНЕЕ СТЕКЛО -----
    sor {
        13,
        <0.0, 0.0>,
        <1.2, 0.0>,
        <1.4, 0.5>,
        <1.6, 1.5>,
        <1.6, 5.5>,
        <1.5, 7.0>,
        <1.2, 8.0>,
        <0.9, 8.8>,
        <0.75, 9.5>,
        <0.75, 12.0>,
        <0.8, 12.3>,
        <0.7, 12.5>,
        <0.0, 12.7>
        open
        texture {
            pigment { color rgbt <1, 1, 1, 0.95> }
            finish {
                reflection { 0.15 }
                specular 0.85
                roughness 0.02
            }
        }
        interior { ior 1.15 }
    }

    // ----- СИРОП (ВНУТРЕННЯЯ ЖИДКОСТЬ) -----
    // Обратите внимание: он немного меньше внешнего, чтобы избежать артефактов наложения (coincident surfaces)
    sor {
        8,
        <0.0, 0.15>,
        <1.05, 0.15>,
        <1.25, 0.6>,
        <1.45, 1.6>,
        <1.45, 3.8>,
        <1.35, 4.2>,
        <0.95, 5.0>,
        <0.0, 4.8>
        open
        texture {
            pigment { color SyrupColor filter 0.8 }
            finish { diffuse 0.9 specular 0.3 }
        }
        interior { ior 1.33 } // Вода/сироп
    }

    // ----- КРЫШКА (ВСЕ ЦИЛИНДРЫ СОБРАНЫ В UNION) -----
    union {
        cylinder { <0, 12.4, 0>, <0, 13.0, 0>, 0.9 }
        cylinder { <0, 13.0, 0>, <0, 15.5, 0>, 0.3 }
        cylinder { <0, 13.0, 0>, <0, 13.5, 0>, 0.5 }
        cylinder { <0, 15.5, 0>, <0, 16.0, 0>, 0.5 }
        cylinder { <0, 15.3, 1.5>, <0, 15.6, 1.5>, 0.2 } // Этот у вас смещен по Z, возможно это кнопка спрея
        
        texture { 
            pigment { color White } 
            finish { specular 0.5 roughness 0.05 }
        }
    }

} // Конец union
#end // Конец макроса
