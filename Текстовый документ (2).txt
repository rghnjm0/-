// Круглые пластины на сиденье
#macro RivetOnSeat(x, z)
   union {
      // Основная пластина
      cylinder { 
         <x, -3.72, z>, 
         <x, -3.68, z>, 
         0.12
         texture { 
            pigment { color rgb <0.85, 0.65, 0.2> } 
            finish { specular 0.7 roughness 0.01 }
         }
      }
      // Ободок
      torus { 0.12, 0.015 rotate x*90 translate <x, -3.69, z> texture { pigment { color rgb <0.7, 0.55, 0.15> } } }
      // Центральная точка
      sphere { <x, -3.69, z>, 0.05 texture { pigment { color rgb <0.9, 0.75, 0.3> } finish { specular 0.8 } } }
   }
#end

// 4 заклёпки на сиденье
RivetOnSeat(-7.8, 5.2)
RivetOnSeat(-5.2, 5.2)
RivetOnSeat(-7.8, 9.7)
RivetOnSeat(-5.2, 9.7)
